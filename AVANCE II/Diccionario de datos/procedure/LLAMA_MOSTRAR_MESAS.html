PROCEDURE llama_mostrar_mesas
AS 
    datos1 SYS_REFCURSOR;
    c_mesa_id mesa.mesa_id%TYPE;
    c_capacidad mesa.capacidad%TYPE;
    c_tipo mesa.tipo%TYPE;
    c_estado mesa.estado%TYPE;
    c_ubicacion mesa.ubicacion%TYPE;
BEGIN 
    mostrar_mesas(datos1);
LOOP
FETCH datos1 INTO c_mesa_id,c_capacidad,c_tipo,c_estado,c_ubicacion;
EXIT WHEN datos1%NOTFOUND;
DBMS_OUTPUT.PUT_LINE(c_mesa_id || ' ' || c_capacidad || ' ' || c_tipo || ' ' || c_estado|| ' ' ||c_ubicacion);
END LOOP; 
END;
