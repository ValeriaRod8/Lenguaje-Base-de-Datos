FUNCTION obtener_mesa(p_mesa_id IN NUMBER)
RETURN VARCHAR2
AS
    v_capacidad mesa.capacidad%TYPE;
    v_tipo mesa.tipo%TYPE;
    v_estado mesa.estado%TYPE;
    v_ubicacion mesa.ubicacion%TYPE;
BEGIN
    SELECT capacidad, tipo, estado, ubicacion
    INTO v_capacidad, v_tipo, v_estado, v_ubicacion
    FROM mesa
    WHERE mesa_id = p_mesa_id;

    RETURN 'Capacidad: ' || v_capacidad || ', Tipo: ' || v_tipo || ', Estado: ' || v_estado || ', Ubicación: ' || v_ubicacion;
    DBMS_OUTPUT.PUT_LINE('Capacidad' || v_capacidad || ', Tipo: ' || v_tipo || ', Estado: ' || v_estado || ', Ubicación: ' || v_ubicacion);
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RETURN 'Mesa no encontrada.';
END;
